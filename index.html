function doGet() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("팀별예산"); const data = sheet.getDataRange().getValues(); const headers = data[0].map(h => String(h).trim()); const jsonArray = []; for (let i = 1; i < data.length; i++) { let row = data[i]; let obj = {}; headers.forEach((header, index) => { obj[header] = row[index]; }); if(obj['연구 주제']) { jsonArray.push(obj); } } return ContentService.createTextOutput(JSON.stringify(jsonArray)) .setMimeType(ContentService.MimeType.JSON); }
